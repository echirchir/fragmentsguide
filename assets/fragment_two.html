public class FragmentTwo extends Fragment
{

  private TextView details;

  @Override
  public View onCreateView(LayoutInflater inflater, ViewGroup container, Bundle state)
  {
   View view = inflater.inflate(R.layout.item_details, container, false);

   details = (TextView)view.findViewById(R.id.details);
   return view;
  }

  public void updateDetails(int position) {
   /*update text view here*/
   String newsDetails = "SELECT details FROM newsfeed WHERE id = position";
   details.setText( newsDetails );
  }

}
